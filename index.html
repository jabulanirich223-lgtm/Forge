local Players = game:GetService("Players")

-- Config (editable via RBXFORGE dashboard)
local CONFIG = {
  patrolRadius = 20,
  chaseRadius = 40,
  attackRadius = 5,
  moveSpeed = 12,
  maxHealth = 100,
  attackDamage = 15,
  attackCooldown = 1.5,
}

local EnemyModule = {}
EnemyModule.__index = EnemyModule

function EnemyModule.new(model, config)
  local self = setmetatable({}, EnemyModule)
  self.model = model
  self.config = config or CONFIG
  self.state = "patrol"
  self.health = self.config.maxHealth
  self.origin = model.HumanoidRootPart.Position
  self.lastAttack = 0
  return self
end

function EnemyModule:getClosestPlayer()
  local closest, minDist = nil, math.huge
  for _, player in ipairs(Players:GetPlayers()) do
    if player.Character then
      local dist = (player.Character.HumanoidRootPart.Position
        - self.model.HumanoidRootPart.Position).Magnitude
      if dist < minDist then
        closest, minDist = player, dist
      end
    end
  end
  return closest, minDist
end

function EnemyModule:update(dt)
  local player, dist = self:getClosestPlayer()
  local humanoid = self.model:FindFirstChild("Humanoid")
  
  if dist < self.config.attackRadius then
    self.state = "attack"
    self:attack(player)
  elseif dist < self.config.chaseRadius then
    self.state = "chase"
    humanoid:MoveTo(player.Character.HumanoidRootPart.Position)
  else
    self.state = "patrol"
    self:patrol()
  end
end

return EnemyModule
